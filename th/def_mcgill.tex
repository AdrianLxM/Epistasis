% Define \mytable
% Include one of my tables, in the standard way
%  Parm 1 is the table name
%  Parm 2 is the caption for the "List of tables"
%  Parm 3 is the real caption
\newcommand{\mytable}[3]{
    \begin{table}[htbp]
        \begin{minipage}{\textwidth}
          \begin{center}
	          \TableCaptionOpt{#2 \label{tab:#1}}{#3}
			  \label{tab:#1}
              \input{tables/#1/table.tex}
          \end{center}
        \end{minipage}
    \end{table}
    \normalsize
}

% Define \fig
% Include one of my figures, in the standard way
%  Parm 1 : File name (no extension)
%  Parm 2 : Width
%  Parm 3 : Caption
\newcommand{\fig}[4]{
\begin{figure}[ht]
    \begin{center}
        \begin{minipage}{#2}
            \includegraphics[ width={#2} ]{figs/#1}
            \FigureCaptionOpt{#3}{#4}
            \label{f:#1}
        \end{minipage}
    \end{center}
\end{figure}
}
